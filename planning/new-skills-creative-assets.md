## OG Image System Design

### Concept: Skill-Specific Social Previews

When sharing a specific skill, artifact, or page, the OG image should reflect THAT content, not just the generic site image.

### Implementation Approach

```typescript
// generateSkillOgImage.ts
interface OgImageConfig {
  skillName: string;
  heroImagePath: string;
  description: string;
  category?: string;
}

async function generateSkillOgImage(config: OgImageConfig): Promise<string> {
  const { skillName, heroImagePath, description, category } = config;

  // Use ImageMagick (same as existing OG generator)
  const command = `
    magick
      -size 1200x630 xc:none

      # Left half: skill hero image
      \\( "${heroImagePath}" -resize 600x630^ -gravity center -extent 600x630 \\)
      -gravity west -composite

      # Right half: gradient background
      \\( -size 600x630 gradient:'#1a1a2e'-'#16213e'
         -rotate 90 \\)
      -gravity east -composite

      # Skill name text
      -gravity east
      -fill white
      -font "Press-Start-2P-Regular"
      -pointsize 24
      -annotate +40+200 "${skillName}"

      # Description text
      -fill '#B0B0C0'
      -pointsize 12
      -annotate +40+240 "${description.slice(0, 60)}..."

      # Bottom branding bar
      \\( -size 1200x50 xc:'rgba(20,20,40,0.95)' \\)
      -gravity south -composite

      -fill white
      -pointsize 14
      -gravity south
      -annotate +0+18 "SOME CLAUDE SKILLS  •  someclaudeskills.com"

      "${outputPath}"
  `;

  return outputPath;
}
```

### Script: Generate All Skill OG Images

```bash
#!/bin/bash
# generate-all-skill-og-images.sh

for skill_dir in .claude/skills/*/; do
  skill_name=$(basename "$skill_dir")
  hero_image="website/static/img/skills/${skill_name}-hero.png"

  if [ -f "$hero_image" ]; then
    echo "Generating OG image for $skill_name..."
    node scripts/generateSkillOgImage.js \
      --skill "$skill_name" \
      --hero "$hero_image"
  else
    echo "No hero image found for $skill_name"
  fi
done
```
